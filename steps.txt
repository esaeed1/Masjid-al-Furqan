sudo apt update
sudo apt install nodejs


nano autostart.sh

#!/bin/bash

# Change to the directory of your Node.js application
cd /home/admin/Desktop/PrayerTimingDisplay

# Run your Node.js application
node app.js &

# Wait for a few seconds for the Node.js application to start
sleep 5

# Launch Chromium browser in kiosk mode
chromium-browser --kiosk http://localhost:4000


chmod +x autostart.sh

mkdir -p ~/.config/autostart

cd ~/.config/autostart/

nano autostart.desktop


[Desktop Entry]
Type=Application
Exec=/home/admin/autostart.sh

sudo reboot
